\small
\begin{itemize}[leftmargin=*, parsep=0pt, topsep=0pt, itemsep=0pt]
				\item Datacenter management
				\item Enable usage of existing \ac{AWS} products and solutions within \ac{AWS} Elastic Beanstalk
				\item Provide well-known software stacks 
				\item Integration of further programming languages, databases, and additional services
				\item Efficient load balancing und auto-scaling rules as well as adaptabilities

	\small
	\begin{itemize}[leftmargin=*, parsep=0pt, topsep=0pt, itemsep=0pt]

					\begin{itemize}[leftmargin=*, topsep=0pt, itemsep=0pt]
						\item Amazon Elastic Compute Cloud (Amazon EC2)
						\item Amazon Simple Storage Service (Amazon S3) 
						\item Elastic Load Balancing
						\item Amazon Relational Database Service (Amazon RDS)
						\item Amazon DynamoDB
						\item Amazon SimpleDB
					\end{itemize}
				\item No additional fees for \ac{AWS} Elastic Beanstalk\vspace{-\baselineskip} 
	\end{itemize}
