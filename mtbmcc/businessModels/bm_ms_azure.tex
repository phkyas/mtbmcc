% PROFIT FORMULA
\multicolumn{4}{|L{\columnF}|}{\large \textbf{Profit Formula}
	\small
	\begin{itemize}[leftmargin=*, parsep=0pt, topsep=0pt, itemsep=0pt]
				\item Transaction based revenue stream based on a complex metric system (see a selection below):
					\begin{itemize}[leftmargin=*, topsep=0pt, itemsep=0pt]
						\item Support: Basic (free), Developer, Standard, Professional
						\item SQL Database: per GB (degressive) and number of databases
						\item SQL Reporting: per reporting hours
						\item Redundant Storage (geographical): per GB
						\item Redundant Storage (local): per GB and storage transactions (in millions)
						\item Bandwidth (out, Nord America and Europe): per GB
						\item Bandwidth (out, other locations): per GB
						\item Virtual Windows Computer (XS - XL)
						\item Virtual Linux Computer (XS - XL)
						\item Service Bus: Messages (in Millions) and relay hours
						\item Media Services
						\item Virtual Network: per hour
						\item Active Directory: per Access Control-Transactions (in millions)
						\item Content Delivery Network: per GB and per transactions (in millions)
					\end{itemize}
				\item Revenue sharing (20\%) for applications provided via the Windows Azure Marketplace\vspace{-\baselineskip} 
	\end{itemize}
} \\
