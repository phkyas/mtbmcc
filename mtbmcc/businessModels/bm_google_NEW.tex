	\footnotesize 
	\begin{itemize}[leftmargin=*, parsep=0pt, topsep=0pt, itemsep=0pt]
				\item Free version with limited daily quota (no quota extension possible) and available features
				\item Paid application: USD 9 / Monthly (per application)
					\begin{itemize}[leftmargin=*, topsep=0pt, itemsep=0pt]
						\item Free quota, afterwards transaction-based pricing (see a selection below):
							\begin{itemize}[leftmargin=*, topsep=0pt, itemsep=0pt]
								\item On-demand Frontend Instances: USD 0.08/ Hour
								\item Reserved Frontend Instances: USD 0.05/ Hour
								\item High Replication Datastore: USD 0.24/ \ac{GB}/ Monthly
								\item Outgoing Bandwidth: USD 0.12/ \ac{GB}
								\item Mail \ac{API}: USD 0.01 / 100 recipients
							\end{itemize}				
					\end{itemize}
				\item Premier account for organizations including additional support: USD 150 / Monthly (exclusive resource usage)\vspace{-\baselineskip} 
	\end{itemize}
