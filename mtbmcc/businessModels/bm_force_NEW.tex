% KEY RESOURCES
\multicolumn{2}{L{\columnT}}{ \textbf{Key Resources}
\small
\begin{itemize}[leftmargin=*, parsep=0pt, topsep=0pt, itemsep=0pt]
				\item Infrastructure (datacenter) for hosting the Salesforce.com Force.com platform, Salesforce.com's \ac{SaaS} portfolio, as well as the third-party applications available within the AppExchange marketplace
				\item Salesforce.com's product portfolio, including Sales Cloud, Service Cloud, Marketing Cloud, Force.com, Chatter, Data.com, and Work.com 
				\item Force.com \ac{IDE} (available as standalone application or Eclipse plugin)
				\item Mobile integration libraries for iOS, Android, HTML5, and Hybrid
				\item Language integration libraries for PHP, Adobe AIR and Flex, Java, .NET, as well as Ruby
				\item Cloud integration tools for \ac{AWS}, Microsoft Windows Azure, Google App Engine, and Facebook
				\item AppExchange marketplace with more than 1.770 applications
				\item Consulting partners, for instance accenture\vspace{-\baselineskip}

% PROFIT FORMULA
\multicolumn{4}{L{\columnF}}{ \textbf{Profit Formula}
	\small
	\begin{itemize}[leftmargin=*, parsep=0pt, topsep=0pt, itemsep=0pt]
				\item Platform subscription:
					\begin{itemize}[leftmargin=*, topsep=0pt, itemsep=0pt]
						\item Three basic editions: EUR 15/ 54/ 80/ Monthly (per user)
						\item Additional services are charged separately, for instance Data.com
					\end{itemize}
				\item Support services/ packages
				\item Training services for different roles, ranging from developers to end users: USD 1.800 - 4.000
				\item Certification processes for administrators, developers, implementation experts, and architects
				\item Revenue sharing:
					\begin{itemize}[leftmargin=*, topsep=0pt, itemsep=0pt]
						\item 15\%, applications sold to Salesforce.com customers
						\item 25\%, applications sold to any customer
					\end{itemize}
				\item Onetime fee USD 300 for free apps (security review)\vspace{-\baselineskip} 
	\end{itemize}
