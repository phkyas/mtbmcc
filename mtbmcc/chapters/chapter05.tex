In order to map the values of the function $f(s) \in [a_{1},a_{2}]$ to the desired interval $[b_{1},b_{2}]$, the following generic linear transformation function is used:
\begin{equation}\label{eq:ltg}
	f(t) = b_{1} + \frac{(s-a_{1})(b_{2}-b_{1})}{(a_{2}-a_{1})} \in [b_{1},b_{2}]
\end{equation}

In the particaular case of mapping the function $f(s) \in [0,1]$ to the interval $[0,b_{2}]$, the Formula \ref{eq:ltg} can be simplified as follows:
\begin{equation}\label{eq:lts}
	%f(t) = b_{1} + s (b_{2}-b_{1}) \in [b_{1},b_{2}].
	f(t) = sb_{2} \in [0,b_{2}].
\end{equation}

Platform Improvements:
\begin{equation}
	PI = \frac{Customer Base}{\sum\mathit{Potential~<Customer Segment>}}*\mathit{Transformation}
\end{equation}

\begin{equation}
	\begin{split}
		\mathit{GOV~for~ITS} = &( DC * \mathit{Impact~of~DC~on~the~GOV~for~ITS} +\\
													& AbI * \mathit{Impact~of~AbI~on~the~GOV~for~ITS} +\\
													& DEV * \mathit{Impact~of~DEV~on~the~GOV~for~ITS} +\\
													& INT * \mathit{Impact~of~INT~on~the~GOV~for~ITS}) * GOV_{i}
	\end{split}
\end{equation}

\begin{equation}
	\begin{split}
		\mathit{TS~for~ITS} = &( DC * \mathit{Impact~of~DC~on~the~TS~for~ITS} +\\
													& AbI * \mathit{Impact~of~AbI~on~the~TS~for~ITS} +\\
													& DEV * \mathit{Impact~of~DEV~on~the~TS~for~ITS} +\\
													& INT * \mathit{Impact~of~INT~on~the~TS~for~ITS}) * TS_{i} * PI
	\end{split}
\end{equation}

\begin{equation}
	\begin{split}
		\mathit{AS~for~ITS} = &( DC * \mathit{Impact~of~DC~on~the~AS~for~ITS} +\\
													& AbI * \mathit{Impact~of~AbI~on~the~AS~for~ITS} +\\
													& DEV * \mathit{Impact~of~DEV~on~the~AS~for~ITS} +\\
													& INT * \mathit{Impact~of~INT~on~the~AS~for~ITS}) * AS_{i} * PI
	\end{split}
\end{equation}

\begin{equation}
	\begin{split}
		\mathit{CVP~for~ITS} = &( DC * \mathit{Impact~of~DC~on~the~CVP~for~ITS} +\\
													& AbI * \mathit{Impact~of~AbI~on~the~CVP~for~ITS} +\\
													& DEV * \mathit{Impact~of~DEV~on~the~CVP~for~ITS} +\\
													& INT * \mathit{Impact~of~INT~on~the~CVP~for~ITS}) *\\
													& CVP\_ITS_{i} * PI * AC\_MP
	\end{split}
\end{equation}

\begin{equation}
	\begin{split}
		\mathit{Adoption~Rate~ITS} = &~ \mathit{GOV~for~ITS} + \mathit{TS~for~ITS} +\\
																& \mathit{AS~for~ITS} + \mathit{CVP~for~ITS}		
	\end{split}
\end{equation}

\begin{equation}
	\mathit{Adoption~from~VP~ITS} = \mathit{Potential~ITS} * \mathit{Adoption~Rate~ITS}
\end{equation}

\begin{equation}
	\mathit{Contacts~with~ITS} = \frac{\mathit{Population~ITS}}{\mathit{Population~ITS}+\mathit{Potential~ITS}}
\end{equation}

\begin{equation}
	\mathit{Imitators~ITS} = \mathit{Contacts~with~ITS} * \mathit{Probability~of~Adoption~after~Contact~ITS}
\end{equation}

\begin{equation}
	\mathit{New~ITS} = \mathit{Adoption~from~VP~ITS} + \mathit{Imitators~ITS}
\end{equation}
