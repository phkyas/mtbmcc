% ****************************************************************************************************
% Graph TS and AS
% ****************************************************************************************************

\begin{tikzpicture}[x=0.1cm,y=50cm]

  \def\xmin{0}
  \def\xmax{65}
  \def\ymin{1.0}
  \def\ymax{1.105}

  % grid
  \draw[style=help lines, ystep=0.01, xstep=10] (\xmin,\ymin) grid
  (\xmax,\ymax);

  % axes
  \draw[->] (\xmin,\ymin) -- (\xmax,\ymin) node[right] {\footnotesize $Time~in~Months$};
  \draw[->] (\xmin,\ymin) -- (\xmin,\ymax) node[above] {\footnotesize $TS/~AS$};

  \foreach \x in {0,10,...,60}
    \node at (\x, \ymin) [below] {\scriptsize \x};
  \foreach \y in {1.00,1.02,1.04,1.06,1.08,1.10}
    \node at (\xmin,\y) [left] {\scriptsize \y};

  %,mark=*,mark size=0.5pt
  \draw[color=blue] plot[smooth] file {simulationData/ts.txt}
		node [right] {\scriptsize $TS$};
	\draw[color=red] plot[smooth] file {simulationData/as.txt}
		node [right] {\scriptsize $AS$};


\end{tikzpicture}
